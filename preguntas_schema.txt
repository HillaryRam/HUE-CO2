CREATE TABLE `preguntas` (
  `pregunta_id` bigint unsigned NOT NULL AUTO_INCREMENT,
  `carta_id` bigint unsigned NOT NULL,
  `texto` varchar(255) COLLATE utf8mb4_unicode_ci NOT NULL,
  `created_at` timestamp NULL DEFAULT NULL,
  `updated_at` timestamp NULL DEFAULT NULL,
  PRIMARY KEY (`pregunta_id`),
  KEY `preguntas_carta_id_foreign` (`carta_id`),
  CONSTRAINT `preguntas_carta_id_foreign` FOREIGN KEY (`carta_id`) REFERENCES `cartas` (`carta_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
